\documentclass[a4paper]{article}

% --- Packages ---
\usepackage[margin=0pt]{geometry} % No margins to allow full bleed
\usepackage{xcolor}               % For color control
\usepackage{atbegshi}             % To hook into every page
\usepackage{ifthen}               % For logic

% --- Logic to Color Pages ---
\AtBeginShipout{%
  \AtBeginShipoutUpperLeft{%
    \ifthenelse{\isodd{\value{page}}}%
    {% 
      % Odd Pages: Fill with Black
      \put(0,-\paperheight){\color{black}\rule{\paperwidth}{\paperheight}}%
    }%
    {% 
      % Even Pages: Do nothing (remains white/blank)
    }%
  }%
}

\begin{document}

% Create 1000 empty pages
\multido{\i=1+1}{1000}{
    \null
    \newpage
}

\end{document}
